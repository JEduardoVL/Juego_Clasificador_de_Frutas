CREATE DATABASE usuarios;
use usuarios;
CREATE TABLE login (
 idLOGIN INT NOT NULL AUTO_INCREMENT,
 USERNAME VARCHAR(45) NULL,
 PASSWORD VARCHAR(45) NULL,
 PRIMARY KEY (idLOGIN)
);
